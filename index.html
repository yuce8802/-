async function fetchProducts() { const grid = document.getElementById('product-grid'); try { // 移除所有参数，只保留最纯粹的请求 const url = `https://api.vika.cn/fusion/v1/datasheets/${SHEET_ID}/records`; const response = await fetch(url, { headers: { 'Authorization': `Bearer ${VIKA_TOKEN}` } }); const result = await response.json(); if (result.success && result.data.records.length > 0) { allProducts = result.data.records; console.log("API 原始数据:", allProducts[0].fields); // 在控制台查看真实的列名是什么 updateStats(); updateContent(); } else { grid.innerHTML = `
SHEET IS EMPTY
`; } } catch (e) { grid.innerHTML = `
FETCH FAILED
`; } } function renderUI() { const grid = document.getElementById('product-grid'); if (!allProducts || allProducts.length === 0) return; grid.className = viewMode === 0 ? "grid grid-cols-2 md:grid-cols-3 gap-4 md:gap-8" : viewMode === 1 ? "mode-stack" : "mode-ring"; grid.innerHTML = allProducts.map((r, i) => { const f = r.fields; // --- 💡 智能字段探测器 --- let img = ""; let title = ""; let link = ""; // 遍历这行数据的所有列，按内容特征自动匹配 for (let key in f) { const value = f[key]; const keyName = key.trim(); // 去掉表头可能存在的空格 // 1. 找图片：如果是数组或是包含 cloudinary/http 的字符串 if (!img) { if (Array.isArray(value) && value[0]?.url) img = value[0].url; else if (typeof value === 'string' && (value.includes('http') || value.includes('cloudinary'))) img = value; } // 2. 找标题：匹配“标题”、“Title”或者第一个非 URL 的长文本 if (!title) { if (keyName.includes('标题') || keyName.includes('Title') || keyName.includes('Name')) title = value; } // 3. 找链接：包含 amazon/http 且不是刚才那张图 if (!link && typeof value === 'string' && value.includes('http') && value !== img) { link = value; } } // 兜底处理 img = (img || "").replace("http://", "https://"); title = title || "VortinT Item"; link = link || "#"; const angle = (360 / allProducts.length) * i; const style = viewMode === 2 ? `--ry: ${angle}deg;` : ''; return `

${title}

Explore →
`; }).join(''); }
