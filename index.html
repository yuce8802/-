"mode-stack" : "mode-ring"; grid.innerHTML = allProducts.map((r, i) => { const f = r.fields; let img = ""; let title = "VortinT Item"; let link = "#"; // 💡 暴力扫描：不准列名，只抓内容 Object.values(f).forEach(val => { const str = String(Array.isArray(val) ? (val[0]?.url || "") : val); if (str.includes('cloudinary.com') || (str.includes('http') && (str.endsWith('.jpg') || str.endsWith('.png') || str.endsWith('.webp')))) { img = str; } else if (str.includes('amazon.com') || str.includes('amzn.to')) { link = str; } else if (typeof val === 'string' && val.length > 2 && !val.includes('http')) { title = val; } }); // 强制 HTTPS img = img.replace("http://", "https://"); const angle = (360 / allProducts.length) * i; const style = viewMode === 2 ? `--ry: ${angle}deg;` : ''; return `

${title}
